1 /**
  2     An object that causes the level to end when it it touched
  3     @author <a href="mailto:thierry.prost@gmail.com">Thierry PROST</a>
  4     @class
  5 */
  6 function LevelEndPost()
  7 {
  8 
  9     this.startupLevelEndPost = function(/**Image*/ image, /**Number*/ x, /**Number*/ y, /**Number*/ z)
 10     {
 11         this.startupAnimatedGameObject(image, x, y, z, 4, 10);
 12         return this;
 13     }
 14 
 15     this.shutdown = function()
 16     {
 17         this.shutdownLevelEndPost();
 18     }
 19 
 20     this.shutdownLevelEndPost = function()
 21     {
 22         this.shutdownAnimatedGameObject();
 23     }
 24 
 25     /**
 26         Updates the object
 27         @param dt The time since the last frame in seconds
 28         @param context The drawing context
 29         @param xScroll The global scrolling value of the x axis
 30         @param yScroll The global scrolling value of the y axis
 31     */
 32 	this.update = function (/**Number*/ dt, /**CanvasRenderingContext2D*/context, /**Number*/ xScroll, /**Number*/ yScroll)
 33     {
 34         if (this.collisionArea().intersects(g_player.collisionArea()))
 35         {
 36             g_ApplicationManager.openMainMenu();
 37             this.shutdown();            
 38         }
 39     }
 40 }
 41 LevelEndPost.prototype = new AnimatedGameObject;
